  incrementElement(a, b) {
    if (a.length !== b.length) {
        throw new Error("Les listes a et b doivent avoir la mme longueur");
    }

    let c = [...b]; // Cre une copie de la liste b
    for (let i = 0; i < a.length; i++) {
        if (a[i] === 1) {
            c[i] = b[i] + 1; // Incrmente l'lment i-me de b
            break; // Sort de la boucle aprs avoir trouv l'lment gal  1
        }
    }

    return c;
}
